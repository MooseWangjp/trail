# 闭包

## JavaScript为什么需要闭包

JavaScript的函数是“一等公民”，这就意味着

1. 函数可以给变量赋值

```javascript
const fn = function () {
  console.log("Hello World");
};
```

2. 函数可以作为参数（callback）

```javascript
function execute(callback) {
  callback();
}

execute(function () {
  console.log("executed");
});
```

3. 函数可以作为返回值

```javascript
function creator() {
  return function () {
    console.log("returned");
  };
}
```

这些和闭包有什么关系？

因为，一旦函数可以作为返回值，闭包就是“自然而然”的结果。因为返回的函数需要能够访问它原来作用域中的变量。

那么反过来想一下，如果JavaScript没有实现闭包，它会怎么样？我们用几个场景来看一下效果：

【场景1】：事件处理

======= 有闭包 =========

```javascript
function setupButton(userId){
    document.getElementById('submit').onclick = function{
        // 这个回调函数记住了userId
        sendRequest(userId)
    }
}
```

====== 没有闭包 ========

```javascript
function setupButton(userId) {
  // 必须把数据存储在全局或DOM上
  document.getElementById("submit").dataset.userId = userId;

  document.getElementById("submit").onclick = function () {
    // 然后从DOM重新获取数据
    const userId = this.dataset.userId;
    sendRequest(userId);
  };
}
```

结论：没有闭包的情况下，异步编程会变得非常繁琐，特别是在参数多了之后

【场景2】：模拟私有变量

======= 有闭包 =========

```javascript
function createBankAccount(initialBalance) {
  let balance = initialBalance; // 私有

  return {
    deposit: function (amount) {
      balance += amount;
    },
    getBalance: function () {
      return balance;
    },
  };
}
```

====== 没有闭包 ========

```javascript
function createBankAccount(initialBalance) {
  // 必须使用对象，但所有属性都是可访问的
  return {
    balance: initialBalance, // 需要作为公共的变量供deposit使用，不是私有的！
    deposit: function (amount) {
      this.balance += amount;
    },
  };
}
// 任何人都可以直接修改 balance
account.balance = 9999999;
```

结论：没有闭包，JavaScript永远无法实现真正的数据封装

还有很关键的一点事，JavaScript的执行模型要求有闭包，我们都知道JavaScript的执行模式是单线程+事件循环，异步操作的回调函数是在未来某个时间触发执行的，那么如果没有闭包，回调函数在执行的时候就无法获取到当前函数体内定义的变量。

所以我们回到闭包的核心概念，讲的是：

> 当一个函数能够记住并访问其词法作用域，即使该函数在其词法作用域之外执行时也能获取到作用域内的变量，这种情况下就形成了闭包。

## 形成闭包的必要条件

- 函数嵌套+变量访问：必须有嵌套的函数，且内部函数访问外部函数的变量

```javascript
// ✅ 形成闭包
function outer() {
  let x = 10;
  function inner() {
    return x + 1; // 访问了外部变量 x
  }
  return inner;
}

// ❌ 不形成闭包（没有访问外部变量）
function outer() {
  let x = 10;
  function inner() {
    return 1 + 1; // 没有访问 x
  }
  return inner;
}

// ❌ 不形成闭包（没有嵌套）
let x = 10; // 全局变量，不是闭包
function inner() {
  return x + 1;
}
```

- 内部函数被传递到外部：内部函数必须被返回、赋值或传递到外部

```javascript
// ✅ 形成闭包 - 返回内部函数
function outer() {
  let x = 10;
  return function inner() {
    return x;
  };
}
const fn = outer();

// ✅ 形成闭包 - 赋值给外部变量
function outer() {
  let x = 10;
  let inner = function () {
    return x;
  };
  globalCallback = inner; // 传递到全局
}

// ✅ 形成闭包 - 作为参数传递
function setup() {
  let config = { debug: true };
  setTimeout(function callback() {
    console.log(config); // 传递给 setTimeout
  }, 1000);
}

// ❌ 不形成闭包 - 内部函数未被传出
function outer() {
  let x = 10;
  function inner() {
    return x;
  }
  // inner 只在内部调用，没有被传出
  inner();
}
// outer 执行完毕后，x 被销毁
```

- 内部函数被外部调用：内部函数必须在外部作用域被调用

```javascript
// ✅ 形成闭包 - 在外部调用
function outer() {
  let x = 10;
  return function () {
    return x;
  };
}
const fn = outer();
console.log(fn()); // 在外部调用

// ⚠️ 技术上形成了闭包对象，但立即被垃圾回收
function outer() {
  let x = 10;
  return function () {
    return x;
  };
}
outer(); // 返回的函数没有被保存，立即被GC
```

总结：判断是否形成闭包的标准其实是判断 **作用域链是否跨越了函数边界**

## 闭包的好处

1. 数据封装和私有化

```javascript
function createUser(name) {
  let _name = name; // 私有变量

  return {
    getName: () => _name,
    setName: (newName) => {
      _name = newName;
    },
  };
}

const user = createUser("张三");
console.log(user.getName()); // '张三'
console.log(user._name); // undefined（无法直接访问）
```

2. 保持状态

```javascript
function makeAdder(x) {
  // x 被闭包"记住"了
  return function (y) {
    return x + y;
  };
}

const add5 = makeAdder(5);
const add10 = makeAdder(10);

console.log(add5(3)); // 8
console.log(add10(3)); // 13
```

3. 函数工厂

```javascript
function createLogger(prefix) {
  return function (message) {
    console.log(`[${prefix}] ${message}`);
  };
}

const infoLogger = createLogger("INFO");
const errorLogger = createLogger("ERROR");

infoLogger("操作成功"); // [INFO] 操作成功
errorLogger("操作失败"); // [ERROR] 操作失败
```

4. 异步编程种的回调保持

```javascript
function fetchData(url) {
  let cachedData = null;

  return function () {
    if (cachedData) {
      return Promise.resolve(cachedData);
    }
    return fetch(url).then((data) => {
      cachedData = data; // 闭包保存缓存
      return data;
    });
  };
}
```

闭包给JavaScript带来了很多优势，给JS提供了更多的拓展可能，但是同时，不合理地使用也会带来大量问题

## 闭包的问题

1. 内存泄漏风险

```javascript
//  问题代码
function setupHandlers() {
  const hugeData = new Array(1000000).fill("data");

  document.getElementById("button").onclick = function () {
    console.log("clicked", hugeData);
    // 这个闭包会一直保持对 hugeData 的引用
    // 即使只用到了简单的 console.log
  };
}

//  解决方案：只捕获需要的数据
function setupHandlers() {
  const hugeData = new Array(1000000).fill("data");
  const importantValue = hugeData[0]; // 只提取需要的值

  document.getElementById("button").onclick = function () {
    console.log(importantValue);
  };
  // hugeData 现在可以被垃圾回收
}
```

2. 循环中的闭包问题

```javascript
// 问题代码
for (var i = 0; i < 3; i++) {
  setTimeout(function () {
    console.log(i); // 输出: 3, 3, 3（不是 0, 1, 2）
  }, 100);
}

// 解决方案1：使用 let
for (let i = 0; i < 3; i++) {
  setTimeout(function () {
    console.log(i); // 输出: 0, 1, 2
  }, 100);
}

// 解决方案2：使用 IIFE
for (var i = 0; i < 3; i++) {
  (function (j) {
    setTimeout(function () {
      console.log(j); // 输出: 0, 1, 2
    }, 100);
  })(i);
}
```

3. 性能问题

```javascript
// 每次调用都创建新的闭包
function inefficient() {
  const bigArray = new Array(10000);

  return {
    method1: function () {
      /* 使用 bigArray */
    },
    method2: function () {
      /* 使用 bigArray */
    },
    method3: function () {
      /* 使用 bigArray */
    },
  };
}

//  更好的方式：共享闭包
function efficient() {
  const bigArray = new Array(10000);

  // 方法放在原型上，共享数据
  const obj = {
    getData: () => bigArray,
  };
  Object.setPrototypeOf(obj, efficient.prototype);
  return obj;
}
```
